# Python dependencies for the Mastermind Abacus automation backend.
#
# Pinned to the versions known to work with this codebase. Use:
#     pip install -r api/requirements.txt
#
# Runtime requirements: Python 3.10+, ffmpeg+ffprobe on PATH (or
# FFMPEG_BIN/FFPROBE_BIN env vars), Bahnschrift/Arial fonts available
# at the paths in scripts/image_generator.py (Windows ships these by
# default; Linux installs need to provide equivalents).

# Web server -----------------------------------------------------------------
Flask==3.1.3
flask-cors==5.0.0
# Production WSGI server. Local dev uses Flask's built-in dev server;
# the Docker image runs both Flask apps under gunicorn behind Traefik.
gunicorn==23.0.0

# Image generator ------------------------------------------------------------
# Used by scripts/image_generator.py (PIL.Image, ImageDraw, ImageFont).
Pillow>=10.4.0,<12.0.0

# Optional — Google Sheets CSV fetch (api/app.py /_fetch_gsheet_csv).
# Code path degrades gracefully if missing; install for the sheet-connect
# feature.
requests>=2.32.0,<3.0.0

# Optional — xlsx upload support for /recipients/upload.
# Code path degrades gracefully if missing.
openpyxl>=3.1.5,<4.0.0

# Notes:
#   - The video generator (scripts/video_generator.py) shells out to
#     ffmpeg + ffprobe; install those separately at the OS level
#     (https://www.gyan.dev/ffmpeg/builds/ for Windows).
#   - No SMTP library needed — auth_server.py uses Python's stdlib
#     smtplib for password-reset email.
